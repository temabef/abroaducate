### **6.3 Scholarship Database Design**

**Supabase Schema:**
```sql
-- Scholarships table
CREATE TABLE scholarships (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title VARCHAR NOT NULL,
  description TEXT,
  amount VARCHAR,
  deadline DATE,
  eligibility TEXT,
  requirements TEXT,
  application_link VARCHAR,
  provider VARCHAR,
  country VARCHAR,
  field_of_study VARCHAR,
  education_level VARCHAR,
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW(),
  is_active BOOLEAN DEFAULT true
);

-- User scholarship tracking
CREATE TABLE user_scholarships (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES auth.users(id),
  scholarship_id UUID REFERENCES scholarships(id),
  status VARCHAR DEFAULT 'interested', -- interested, applying, applied, rejected, awarded
  notes TEXT,
  deadline_reminder BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT NOW()
);
```
